library(ggplot2)

gghist <- function(x){
  if (!is.numeric(x)) {
    stop("Please enter a numeric vector")
  }
  df <- data.frame(x = x)
  
  ggplot(df, aes(x = x)) +
    geom_histogram(
      binwidth =0.7,
      color ="black",
      fill ="lightblue") +
    labs(y ="Count", x ="Input", title ="Histogram")
}
library(ggplot2)

ggbox <- function(x, group =NULL, orientation = c("horizontal", "vertical")){
  
  if (!is.numeric(x)){
    stop("Please enter a numeric vector")
  }
  
  if (is.null(group)){
    y <- data.frame(x = x)
    
    g <- ggplot(y, aes(y = x)) +
      geom_boxplot(fill ="lightgreen") +
      labs(x ="", y ="")
  } else {
    y <- data.frame(x = x, group = group)
    
    g <- ggplot(y, aes(x =group, y =x)) +
      geom_boxplot(fill ="pink") +
      labs(x ="", y="")
  }
  
  if (orientation == "horizontal"){
    g <- g + coord_flip()
  }
  
  return(g)
}

Part j (7 points)

Generalize this by writing a function that will do this for a given team, and create a tibble with this information for all teams. Arrange this tibble first by winning percentage (descending) and by team (alphabetically) if there are any ties when it comes to winning percentage. A for loop and the add_row() function may be useful here.

team_win_record <- function(x){
  games_2025 %>% 
    filter(Visiting_Team == x | Home_Team == x) %>% 
    mutate(
      win = case_when(
        Home_Team == x & Home_Score > Visiting_Score ~ 1,
        Visiting_Team == x & Visiting_Score > Home_Score ~ 1,
        TRUE ~ 0
      )
    ) %>% 
    summarise(
      Team = x,
      Wins = sum(win),
      Losses = n() - Wins,
      Games = n(),
      Win_Percentage = Wins/Games
    )
}

all_team <- sort(unique(c(games_2025$Visiting_Team, games_2025$Home_Team)))

records <- tibble(

for (t in all_team) {
